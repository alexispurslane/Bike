def eval(context)
  name.each do |name|
    if !@@is_set[name]
      context.locals[name] = value.eval(context).call(name, [])
      @@is_set[name] = true
    else
      raise "Attemt to re-assign variable using normal variable."
    end
  end
  Constants["nil"]
end
